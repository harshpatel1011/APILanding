import { NexusClient } from '@nexus/api';

const client = new NexusClient({
  apiKey: process.env.NEXUS_API_KEY,
  environment: 'production'
});

// Process data stream in real-time
const response = await client.data.processStream({
  sourceId: 'src_9kx2p1',
  pipeline: ['clean', 'enrich', 'analyze'],
  options: {
    webhookUrl: 'https://myapp.com/webhook'
  }
});

console.log(`Job started: ${response.jobId}`);
